#include <stdio.h>
int main()
{
    int i,j,k;
    for (i=0;i<500;i=i+1) {
        for (j=1;j<500;j=j+1) {
            for (k=0;k<500;k=k+1) {
                if ((k*k)==(i*i)+(j*j) && j<i) {
                    printf("%d^2+%d^2=%d^2\n",i,j,k);
                }
            }
        }
    }
    return 0;
}
