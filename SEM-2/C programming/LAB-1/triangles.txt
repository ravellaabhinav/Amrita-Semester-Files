#include <stdio.h>
int main()
{
    int a,b,c;
    scanf("%d %d %d",&a,&b,&c);
    if ((a+b)>c && (b+c)>a && (a+c)>b){
        if (a==b && b==c && c==a) {
            printf("equilateral triangle");
        }
        else if (a==b || b==c || c==a) {
            printf ("Isoscles triangle");
        }
        else if (a!=b && b!=c && c!=a) {
            printf("Scalene triangle");
        }
    }
    else {
        printf("Invalid triangle)");
    }
    return 0;
}
